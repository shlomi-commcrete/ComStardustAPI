cmake_minimum_required(VERSION 3.10)
project(rnnoise_jni C)

# ─── Locate xiph/rnnoise source tree ─────────────────────────────────────────
# Expected layout (created by stardust/scripts/setup_rnnoise.sh):
#   stardust/src/main/cpp/rnnoise/
#     ├── include/rnnoise.h
#     └── src/*.c
set(RNNOISE_DIR ${CMAKE_CURRENT_SOURCE_DIR}/rnnoise)

if(NOT EXISTS ${RNNOISE_DIR}/include/rnnoise.h)
    message(FATAL_ERROR
        "RNNoise sources not found at ${RNNOISE_DIR}.\n"
        "Run: ./stardust/scripts/setup_rnnoise.sh\n"
        "to clone xiph/rnnoise and generate the model data file.")
endif()

# ─── Glob src/*.c, excluding the standalone tools and example mains ──────────
file(GLOB RNNOISE_SOURCES CONFIGURE_DEPENDS ${RNNOISE_DIR}/src/*.c)
list(FILTER RNNOISE_SOURCES EXCLUDE REGEX ".*[/\\\\]examples?[/\\\\].*")
list(FILTER RNNOISE_SOURCES EXCLUDE REGEX ".*[/\\\\]dump_data\\.c$")
list(FILTER RNNOISE_SOURCES EXCLUDE REGEX ".*[/\\\\]dump_features\\.c$")
list(FILTER RNNOISE_SOURCES EXCLUDE REGEX ".*[/\\\\]compile\\.c$")
list(FILTER RNNOISE_SOURCES EXCLUDE REGEX ".*[/\\\\]test_.+\\.c$")

if(NOT RNNOISE_SOURCES)
    message(FATAL_ERROR "No RNNoise source files found in ${RNNOISE_DIR}/src")
endif()

# ─── rnn_data.c / rnnoise_data.c is generated by ./configure or ./autogen.sh ─
# The setup script handles that. Verify presence.
set(_HAS_DATA FALSE)
foreach(_src IN LISTS RNNOISE_SOURCES)
    if(_src MATCHES ".*[/\\\\](rnn_data|rnnoise_data)\\.c$")
        set(_HAS_DATA TRUE)
    endif()
endforeach()
if(NOT _HAS_DATA)
    message(FATAL_ERROR
        "Model data file (rnn_data.c or rnnoise_data.c) not found.\n"
        "The setup script should have generated it. Re-run:\n"
        "  ./stardust/scripts/setup_rnnoise.sh")
endif()

include_directories(
    ${RNNOISE_DIR}/include
    ${RNNOISE_DIR}/src
)

add_library(rnnoise_jni SHARED
    rnnoise_jni.c
    ${RNNOISE_SOURCES}
)

# RNNoise expects these defines for stand-alone (non-autotools) builds.
target_compile_definitions(rnnoise_jni PRIVATE
    COMPILE_OPUS=0
)

target_compile_options(rnnoise_jni PRIVATE
    -O3
    -ffast-math
    -fvisibility=hidden
    -DNDEBUG
    -Wno-implicit-function-declaration   # rnnoise sources are tolerant
    -Wno-unused-variable
    -Wno-unused-function
)

# Strip on release builds to keep the .so small.
target_link_options(rnnoise_jni PRIVATE
    "$<$<CONFIG:Release>:-Wl,--gc-sections>"
    "$<$<CONFIG:Release>:-Wl,--strip-all>"
)

find_library(log-lib log)
target_link_libraries(rnnoise_jni ${log-lib})
